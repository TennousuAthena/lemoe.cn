  "Pneumonia" varchar(255),
  "Pneumothorax" varchar(255),
  "Support Devices" varchar(255)
);

(
  "dicom_id" varchar(255),
  "subject_id" int8,
  "study_id" int8,

(

  "Pleural Other" varchar(255),
  "Pneumonia" varchar(255),
  "Pneumothorax" varchar(255),
  "Support Devices" varchar(255)
);

(
  "dicom_id" varchar(255),

  "split" varchar(255)
